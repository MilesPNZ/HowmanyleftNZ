** SQL COMMANDS TO RUN LOCALLY IN DB Browser **
1. This command will take the uploaded vehicles table from the raw NZTA data, create 8 columns and format the data for site use

CREATE TABLE counts_current AS
SELECT
    make,
    model,
    submodel,
    vehicle_year,
    COALESCE(motive_power, alternative_motive_power, '') AS fuel_type,
    class,
    COUNT(*) AS count,
    DATE('now') AS snapshot_date
FROM vehicles
GROUP BY make, model, submodel, vehicle_year, fuel_type, class;

-- Enable trigram extension for fast partial-match searches (ILIKE '%term%')
CREATE EXTENSION IF NOT EXISTS pg_trgm;

-- Core indexes
CREATE INDEX idx_counts_make_model
    ON counts_current (make, model);

CREATE INDEX idx_counts_year
    ON counts_current (vehicle_year);

CREATE INDEX idx_counts_fuel
    ON counts_current (fuel_type);

CREATE INDEX idx_counts_count_desc
    ON counts_current (count DESC);

-- Trigram indexes — make ILIKE super fast
CREATE INDEX idx_counts_make_model_trgm
    ON counts_current USING GIN ((make || ' ' || model) gin_trgm_ops);

CREATE INDEX idx_counts_model_trgm
    ON counts_current USING GIN (model gin_trgm_ops);
